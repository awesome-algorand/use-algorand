<Tabs>
    <CodeTabItem label="algosdk">
        #### Install
        ```bash
        npm install algosdk --save
        ```
        #### Usage
        ```js
        import { Algodv2 } from "algosdk";

        const client = new Algodv2( "", "https://testnet-api.algonode.cloud", 443 );
        const result = await client.getTransactionParams().do();
        ```

### <IconTitle name="seti:react">React</IconTitle>
<Tabs>
    <CodeTabItem label="@awesome-algorand/react-query">
        #### Install
        ```bash
        npm install @awesome-algorand/react-query @txnlab/use-wallet-react @tanstack/react-query algosdk --save
        ```
        #### Usage
        ```jsx
        import { useTransactionParams } from '@awesome-algorand/react-query';

        export function TransactionParamsViewer() {
            const query = useTransactionParams();
        }
        ```

    <CodeTabItem label="@tanstack/react-query">
        <Aside type="caution" slot="before-code">
            Using the query-core library directly is not recommended!
        </Aside>
        #### Install
        ```bash
        npm install @awesome-algorand/query-core @txnlab/use-wallet-react @tanstack/react-query algosdk --save
        ```
        #### Usage
        ```jsx
        import { useQuery } from '@tanstack/react-query';
        import { transactionParams } from '@awesome-algorand/query-core';
        import { useWallet } from "@txnlab/use-wallet-react";

        export function PendingTransactionsViewer() {
            const walletManager = useWallet();
            const query = useQuery(transactionParams());
        }
        ```
    </CodeTabItem>
